#fir_24t_cplx
make all TARGET=x86sim DATA_TYPE=cint16 COEFF_TYPE=cint16  FIR_LEN=24 SHIFT=16 INPUT_WINDOW_VSIZE=256
# zigzag arch
make all TARGET=x86sim DATA_TYPE=cint16 COEFF_TYPE=int16  FIR_LEN=36 CASC_LEN=1 SHIFT=16
# incloads arch
make all TARGET=x86sim DATA_TYPE=cint32 COEFF_TYPE=cint32 FIR_LEN=10 CASC_LEN=1 SHIFT=16
# basic arch
make all TARGET=x86sim DATA_TYPE=cint32 COEFF_TYPE=cint32 FIR_LEN=10 CASC_LEN=1 SHIFT=16 INPUT_WINDOW_VSIZE=240 

#fir_24t_cplx case USE_COEFF_RELOAD
make all TARGET=x86sim DATA_TYPE=cint16 COEFF_TYPE=cint16  FIR_LEN=24 CASC_LEN=1 SHIFT=16 INPUT_WINDOW_VSIZE=256 USE_COEFF_RELOAD=1

#cascade case
make all TARGET=x86sim DATA_TYPE=cint16 COEFF_TYPE=cint16  FIR_LEN=48 CASC_LEN=3 SHIFT=16

#coefficient reload with cascades
make all TARGET=x86sim DATA_TYPE=cint16 COEFF_TYPE=int16 FIR_LEN=16 CASC_LEN=2 SHIFT=16 INPUT_WINDOW_VSIZE=256 USE_COEFF_RELOAD=1
